function getMongoUrl(config, withCredentials) {
  if (config.url) {
    return config.url;
  } else {
    return 'mongodb://'
       + (withCredentials ? (config.user ? config.user + ':' + config.password + '@' : '') : '')
       + config.host
       + (config.port ? ':' + config.port : '')
       + '/' + config.db;
  }
}

module.exports = dbService;
